SELECT 
  s.student_id, s.full_name, s.fellowship_code,
  s.class_option_id,
  MAX(CASE WHEN sms.milestone_code = 'BORN_AGAIN' 
    THEN sms.completed::text ELSE 'false' END) as BORN_AGAIN,
  MAX(CASE WHEN sms.milestone_code = 'WATER_BAPTIZED' 
    THEN sms.completed::text ELSE 'false' END) as WATER_BAPTIZED,
  MAX(CASE WHEN sms.milestone_code = 'HOLY_SPIRIT' 
    THEN sms.completed::text ELSE 'false' END) as HOLY_SPIRIT
FROM students s
LEFT JOIN student_milestone_status sms ON sms.student_id = s.student_id
WHERE s.class_option_id = $class_option_id
GROUP BY s.student_id, s.full_name, s.fellowship_code, s.class_option_id
ORDER BY s.full_name;
